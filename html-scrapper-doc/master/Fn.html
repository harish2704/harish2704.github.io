var Fn = require('html-scrapper').Fn;

var schema = {
 items: [{
   $rule: 'ul#items li'
   name:{
     $rule: 'a.title',
     $fn: Fn.text
   },
   price:{
     $rule: 'b.price',
     $fn: Fn.asInt
   },
 }],
};
